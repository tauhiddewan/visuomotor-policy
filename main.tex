\section{Introduction}

\begin{frame}{Robotics Research Setting}
\begin{itemize}
    \item Robotics research is typically developed in stages, beginning in simulation and controlled environments before deployment on physical hardware \cite{Kober2013RLRobotics,Tobin2017DomainRandomization}.
    \item This progression is common because simulation enables faster experimentation, safer testing, and easier debugging than direct real-robot development \cite{Kober2013RLRobotics,Tobin2017DomainRandomization}.
    \item However, robotic systems are often studied under predefined objects, layouts, and task assumptions rather than in open-ended human environments \cite{Kasaei2021LifelongServiceRobots,Zheng2025EmbodiedManipSurvey}.
    \item As a result, strong performance in controlled settings does not necessarily imply robustness in longer-term deployment \cite{Tobin2017DomainRandomization,Kasaei2021LifelongServiceRobots}.
\end{itemize}
\end{frame}

\begin{frame}{Visuomotor Manipulation}
\begin{itemize}
    \item A central problem in robotics is \textbf{visuomotor manipulation}: using perception to guide action for tasks such as reaching, grasping, and placing objects \cite{Shridhar2022CLIPort,Shridhar2023PerAct,Zheng2025EmbodiedManipSurvey}.
    \item For humanoid robots, this requires detecting an object, estimating grasp-relevant information, and executing manipulation reliably \cite{Zheng2025EmbodiedManipSurvey}.
    \item Prior work has shown strong progress in visuomotor manipulation, but many systems still assume fixed perception and stable deployment conditions \cite{Shridhar2022CLIPort,Shridhar2023PerAct,Brohan2022RT1,Ghosh2024Octo}.
    \item This makes visuomotor manipulation a useful entry point for studying how perception and action should interact in more adaptive robotic systems \cite{Zheng2025EmbodiedManipSurvey,Kasaei2021LifelongServiceRobots}.
\end{itemize}
\end{frame}

\begin{frame}{Why Fixed Perception Is a Limitation}
\begin{itemize}
    \item In real deployment, environments change over time: layouts vary, uncertainty affects control, and previously unseen objects can appear \cite{Kasaei2021LifelongServiceRobots,Zheng2025EmbodiedManipSurvey}.
    \item Fixed perception modules are therefore a limitation for long-term autonomy, because deployed robots may need to recognize new objects after initial training \cite{Kasaei2021LifelongServiceRobots,Zhou2024CILSurvey}.
    \item Updating perception is itself difficult, since learning new classes can degrade previously learned knowledge through catastrophic forgetting \cite{McCloskey1989Catastrophic,French1999Catastrophic,LiHoiem2016LwF}.
    \item In robotics, this problem is especially important because perception errors or unstable updates can directly affect downstream manipulation \cite{Kasaei2021LifelongServiceRobots,Meng2025RoboticLifelongRL,Zheng2025EmbodiedManipSurvey}.
\end{itemize}
\end{frame}

\begin{frame}{Research Motivation and Scope}
\begin{block}{Long-term research vision}
Develop an integrated visuomotor system that enables a humanoid robot to:
\begin{enumerate}
    \item perceive a scene,
    \item recognize a target object,
    \item adapt object knowledge over time, and
    \item execute robust grasp-and-place behavior.
\end{enumerate}
\end{block}

\vspace{-0.15cm}
\begin{alertblock}{Scope of this study}
Controlled tabletop experiments on the Fourier GR-1, 
with emphasis on continual perception and a manipulation foundation for future perception--action integration.
\end{alertblock}
\end{frame}

\bottomrule

\vspace{-0.1cm}
\textbf{Takeaway:} Continual learning supports adaptive perception, but embodied deployment is still much less mature.
\end{frame}

\textbf{Research Line} & \textbf{What It Establishes} & \textbf{Open Gap} \\
\midrule

Visuomotor manipulation {\scriptsize\cite{Shridhar2023PerAct,Brohan2022RT1,Ghosh2024Octo}}
& Strong task execution and learned policy behavior
& Usually assumes fixed perception and stable object knowledge \\[0.35em]

Continual perception {\scriptsize\cite{Hu2021LoRA,Rebuffi2017iCaRL,Zhou2024CILSurvey}}
& Incremental knowledge update with reduced forgetting
& Rarely evaluated through downstream grasping or placement \\[0.35em]

Embodied lifelong robotics {\scriptsize\cite{Kasaei2021LifelongServiceRobots,Meng2025RoboticLifelongRL}}
& Perception and action should evolve together for long-term autonomy
& A stable perception-to-action interface is still missing \\
\bottomrule
\end{tabularx}
\end{table}

\vspace{-0.1cm}
\textbf{Takeaway:} The central gap is not perception or manipulation alone, but stable coordination between evolving perception and downstream action.
\end{frame}

\begin{frame}{Current Methodological Decomposition}
\begin{itemize}
    \item The overall research direction is a modular \textbf{visuomotor pipeline}.
    \item In the long term, the \textbf{perception track} should update object knowledge over time and provide object-centered information for action.
    \item The \textbf{manipulation track} should use that information to plan and execute grasp-and-place behavior.
    \item In the current study, these two tracks are investigated separately:
    \begin{itemize}
        \item a \textbf{manipulation study} for tabletop pick-and-place control,
        \item a \textbf{perception study} for incremental object understanding.
    \end{itemize}
\end{itemize}

\vspace{0.15cm}
\begin{alertblock}{Reason for this decomposition}
It isolates the two sides of the perception--action interface before full integration.
\end{alertblock}
\end{frame}

    \begin{block}{Goal}
        Develop a simulated tabletop pick-and-place task for the Fourier GR-1, 
        assuming cube position is known.\end{block}

\vspace{-0.15cm}

Figure: Hand-designed state machine used as a structured baseline for tabletop pick-and-place.

\begin{alertblock}{Main limitation}
Relies on fixed assumptions, heuristic thresholds, and manually designed stage logic.
\end{alertblock}

\end{frame}

\begin{frame}{Manipulation Study: RL Reformulation}
\begin{itemize}
    \item To reduce dependence on hand-crafted stage logic, the task is reformulated as a \textbf{reinforcement learning problem}.
    \item Training is performed in a parallel Isaac Lab environment using \textbf{PPO}.
    \item The observation includes end-effector state, cube state, placement target, relative geometry, and arm/hand joint states.
    \item The action consists of incremental arm joint commands and a scalar grip-control signal.
    \item The reward encourages reaching, lifting, transport, and successful placement, with penalties for unstable or unnecessary motion.
\end{itemize}

\vspace{-0.15cm}
    \begin{alertblock}{Methodological role}
        This reformulation tests where scripted control becomes insufficient 
        and where learned control becomes necessary.
    \end{alertblock}
\vspace{0.20cm}
\end{frame}

\begin{frame}{Perception Study: Objective}
\begin{block}{Goal}
Develop an incremental semantic segmentation module that can learn new object classes over time while minimizing catastrophic forgetting.
\end{block}

\begin{itemize}
    \item The perception track is motivated by open-world deployment, where new objects may appear after initial training.
    \item The target is not only recognition accuracy, but also stable object knowledge that can later support downstream manipulation.
    \item The long-term role of this module is to provide object-centered information for a future perception-to-action interface.
\end{itemize}
\end{frame}

\begin{frame}{Perception Study: Dataset and Experimental Setup}
\small
\begin{itemize}
    \item \textbf{Dataset:} YCB-Video
    \begin{itemize}
        \item 21 household object classes,
        \item RGB-D video with pixel-level annotations,
        \item class split into base classes (1--10) and new classes (11--21).
    \end{itemize}
    \item \textbf{Proof-of-concept subset:}
    \begin{itemize}
        \item 5 sequences,
        \item 6,638 usable frames,
        \item 5,974 train / 664 test.
    \end{itemize}
    \item \textbf{Model:} SegFormer-B0 with a modified segmentation head.
    \item \textbf{Adaptation mechanism:} LoRA inserted into selected transformer layers.
    \item \textbf{Metric:} mean Intersection-over-Union (mIoU).
\end{itemize}
\end{frame}

\begin{frame}{Next Steps}
\begin{itemize}
    \item On the perception side, move from offline incremental segmentation toward online novel-class adaptation and robot-level integration \cite{Kasaei2021LifelongServiceRobots,Meng2025RoboticLifelongRL}.
    \item On the manipulation side, improve the RL policy until grasping is more robust to pose variation.
    \item Reduce reliance on fixed offsets and predefined stage assumptions in manipulation.
    \item Integrate the perception and manipulation modules into a single visuomotor pipeline.
    \item After stable simulation performance, transfer the integrated system to the physical Fourier GR-1 robot.
\end{itemize}
\end{frame}

\section{Conclusion}

\begin{frame}{Conclusion}
\begin{itemize}
    \item This work targets a foundational robotics problem: tabletop perception-to-manipulation with a humanoid robot \cite{Zheng2025EmbodiedManipSurvey}.
    \item The current contribution on the manipulation side is a progression from scripted control to RL-based adaptive control.
    \item The perception-side contribution is an incremental segmentation study showing that LoRA + distillation gives the strongest retention/plasticity trade-off in the current setup \cite{LiHoiem2016LwF,Hu2021LoRA}.
    \item Together, these studies establish the basis for a larger integrated visuomotor system.
\end{itemize}
\end{frame}
